# ============================================================================
# CONFIGURACION PARA TOPOLOGIA DE 3 SWITCHES CON RSTP (RED AGRICOLA)
# Esquema de Interconexión (Anillo para redundancia RSTP):
#   - SW-Core [ether21] <---> [ether21] SW-Acc-A
#   - SW-Core [ether22] <---> [ether22] SW-Acc-B
#   - SW-Acc-A [ether23] <---> [ether23] SW-Acc-B (Enlace de Redundancia)
# ============================================================================

# ----------------------------------------------------------------------------
# 1. CONFIGURACION PARA: SW-Core (Switch de Distribución / Core)
# ----------------------------------------------------------------------------
# Limpiar configuración previa si es necesario
/interface bridge add name=bridge_agricola protocol-mode=rstp priority=4096 vlan-filtering=no

# Puertos de Acceso (VLAN 30 - Edge_Mgmt)
/interface bridge port add bridge=bridge_agricola interface=ether13 pvid=30 edge=yes
/interface bridge port add bridge=bridge_agricola interface=ether14 pvid=30 edge=yes

# Puertos Troncales hacia el Router (VLAN 99 / Uplink)
/interface bridge port add bridge=bridge_agricola interface=ether15
/interface bridge port add bridge=bridge_agricola interface=ether16

# Enlaces Troncales hacia los switches de Acceso (Loop RSTP)
/interface bridge port add bridge=bridge_agricola interface=ether21
/interface bridge port add bridge=bridge_agricola interface=ether22

# Declarar VLANs en los Troncales
/interface bridge vlan add bridge=bridge_agricola tagged=ether15,ether16,ether21,ether22 vlan-ids=10
/interface bridge vlan add bridge=bridge_agricola tagged=ether15,ether16,ether21,ether22 vlan-ids=20
/interface bridge vlan add bridge=bridge_agricola tagged=ether15,ether16,ether21,ether22 untagged=ether13,ether14 vlan-ids=30
/interface bridge vlan add bridge=bridge_agricola tagged=ether15,ether16,ether21,ether22 vlan-ids=99

# Activar filtrado de VLANs
/interface bridge set bridge_agricola vlan-filtering=yes


# ----------------------------------------------------------------------------
# 2. CONFIGURACION PARA: SW-Acc-A (Switch de Acceso A)
# ----------------------------------------------------------------------------
# Asignar prioridad de puente estándar (mayor que el core para que el core sea root)
/interface bridge add name=bridge_agricola protocol-mode=rstp priority=8192 vlan-filtering=no

# Puertos de Acceso (VLAN 10 - IoT_Sensores)
/interface bridge port add bridge=bridge_agricola interface=ether1 pvid=10 edge=yes
/interface bridge port add bridge=bridge_agricola interface=ether2 pvid=10 edge=yes
/interface bridge port add bridge=bridge_agricola interface=ether3 pvid=10 edge=yes
/interface bridge port add bridge=bridge_agricola interface=ether4 pvid=10 edge=yes
/interface bridge port add bridge=bridge_agricola interface=ether5 pvid=10 edge=yes
/interface bridge port add bridge=bridge_agricola interface=ether6 pvid=10 edge=yes
/interface bridge port add bridge=bridge_agricola interface=ether7 pvid=10 edge=yes
/interface bridge port add bridge=bridge_agricola interface=ether8 pvid=10 edge=yes

# Puertos de Acceso (VLAN 20 - CCTV_Agricola)
/interface bridge port add bridge=bridge_agricola interface=ether9 pvid=20 edge=yes
/interface bridge port add bridge=bridge_agricola interface=ether10 pvid=20 edge=yes
/interface bridge port add bridge=bridge_agricola interface=ether11 pvid=20 edge=yes
/interface bridge port add bridge=bridge_agricola interface=ether12 pvid=20 edge=yes

# Enlaces Troncales (ether21 al Core, ether23 al SW-Acc-B)
/interface bridge port add bridge=bridge_agricola interface=ether21
/interface bridge port add bridge=bridge_agricola interface=ether23

# Declarar VLANs en los Troncales
/interface bridge vlan add bridge=bridge_agricola tagged=ether21,ether23 untagged=ether1,ether2,ether3,ether4,ether5,ether6,ether7,ether8 vlan-ids=10
/interface bridge vlan add bridge=bridge_agricola tagged=ether21,ether23 untagged=ether9,ether10,ether11,ether12 vlan-ids=20
/interface bridge vlan add bridge=bridge_agricola tagged=ether21,ether23 vlan-ids=30
/interface bridge vlan add bridge=bridge_agricola tagged=ether21,ether23 vlan-ids=99

# Activar filtrado de VLANs
/interface bridge set bridge_agricola vlan-filtering=yes
